\section{Problem D}

\[
     \begin{aligned}
     
        EB =  \sum_{c=1}^{6} c * \frac{1}{6} * \frac{1}{6}  
        = \frac{1}{36}*\sum_{c=1}^{6} c 
        = \frac{1}{36} * 21 = 0.583 \\
        
        E(B)^2 = =  \sum_{c=1}^{6} c^2 * \frac{1}{6} * \frac{1}{6}  
        = \frac{1}{36}*\sum_{c=1}^{6} c^2 
        = \frac{1}{36} * 91 = 2.528 \\
        
        Var(B) =  E(B)^2 - (EB)^2 = 2.528 - 0.583^2 = 2.188
        
     \end{aligned}
\]
